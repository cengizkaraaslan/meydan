// FAZ 3 — Prisma model önerileri
// schema.prisma'ya entegre etmeden önce orchestrator gözden geçirsin.
// User modeline relation alanları eklenmeli (Block, Report'taki ters yönlü relation'lar dahil).

enum ConversationParticipantRole {
  MEMBER
}

enum ReportReason {
  SPAM
  HARASSMENT
  MISLEADING
  OTHER
}

enum ReportStatus {
  OPEN
  REVIEWING
  RESOLVED
  REJECTED
}

// 1:1 sohbet. Grup mesajlaşma desteklenmek istenirse participants count'a kısıt koymadan bırak.
model Conversation {
  id           String   @id @default(cuid())
  // Konuşma katılımcıları arasında tutarlı bir anahtar üretmek için:
  // örn. participantsKey = [userAId, userBId].sort().join(":")
  participantsKey String  @unique
  createdAt    DateTime @default(now())
  updatedAt    DateTime @updatedAt
  lastMessageAt DateTime @default(now())

  messages     Message[]
  participants ConversationParticipant[]

  @@index([lastMessageAt])
}

model ConversationParticipant {
  id             String                       @id @default(cuid())
  conversationId String
  userId         String
  role           ConversationParticipantRole  @default(MEMBER)
  joinedAt       DateTime                     @default(now())
  // Bu kullanıcı için bu sohbetin "okundu" cut-off'u
  lastReadAt     DateTime?

  conversation Conversation @relation(fields: [conversationId], references: [id], onDelete: Cascade)
  user         User         @relation("ConversationParticipantUser", fields: [userId], references: [id], onDelete: Cascade)

  @@unique([conversationId, userId])
  @@index([userId])
}

model Message {
  id             String   @id @default(cuid())
  conversationId String
  senderId       String
  text           String   @db.Text
  createdAt      DateTime @default(now())
  // İnce taneli okundu işareti gerekiyorsa MessageRead tablosu ekle.
  // Şimdilik ConversationParticipant.lastReadAt cut-off'u yeterli.
  editedAt       DateTime?
  deletedAt      DateTime?

  conversation Conversation @relation(fields: [conversationId], references: [id], onDelete: Cascade)
  sender       User         @relation("MessageSender", fields: [senderId], references: [id], onDelete: Cascade)

  @@index([conversationId, createdAt])
  @@index([senderId])
}

// Engelleme — yön önemli. blocker -> blocked yönünde tek satır.
// "İki yönlü engelle" semantiği istenirse iki satır yaratmadan sorgu seviyesinde OR ile kontrol et.
model Block {
  id        String   @id @default(cuid())
  blockerId String
  blockedId String
  createdAt DateTime @default(now())
  reason    String?

  blocker User @relation("BlockerToBlock", fields: [blockerId], references: [id], onDelete: Cascade)
  blocked User @relation("BlockedToBlock", fields: [blockedId], references: [id], onDelete: Cascade)

  @@unique([blockerId, blockedId])
  @@index([blockedId])
}

// Bildirim / şikayet.
model Report {
  id          String       @id @default(cuid())
  reporterId  String
  reportedId  String
  reason      ReportReason
  details     String?      @db.Text
  status      ReportStatus @default(OPEN)
  createdAt   DateTime     @default(now())
  reviewedAt  DateTime?
  reviewerId  String?
  resolution  String?      @db.Text

  reporter User  @relation("ReporterToReport", fields: [reporterId], references: [id], onDelete: Cascade)
  reported User  @relation("ReportedToReport", fields: [reportedId], references: [id], onDelete: Cascade)

  @@index([reportedId, status])
  @@index([reporterId, createdAt])
  @@index([status, createdAt])
}
