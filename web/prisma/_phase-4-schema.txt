// FAZ 4 — Önerilen Prisma şema eklemeleri (Web Push)
// Şu an in-memory tracker (src/lib/push-server.ts) kullanıyoruz.
// Kalıcı depolamaya geçildiğinde aşağıdaki modeller schema.prisma'ya eklenebilir.
// NOT: Bu sadece bir öneri taslağı — schema.prisma'ya dokunulmadı.

model PushSubscription {
  id          String              @id @default(cuid())
  userId      String?             // null → anonim/demo abone
  user        User?               @relation(fields: [userId], references: [id], onDelete: Cascade)
  endpoint    String              @unique
  p256dh      String              // subscription.keys.p256dh
  auth        String              // subscription.keys.auth
  userAgent   String?
  createdAt   DateTime            @default(now())
  updatedAt   DateTime            @updatedAt
  lastSeenAt  DateTime            @default(now())

  prefs       NotificationPref[]

  @@index([userId])
}

model NotificationPref {
  id              String           @id @default(cuid())
  subscriptionId  String
  subscription    PushSubscription @relation(fields: [subscriptionId], references: [id], onDelete: Cascade)
  category        EventCategory    // KONSER | FESTIVAL | ...
  city            String?          // opsiyonel şehir filtresi
  createdAt       DateTime         @default(now())

  @@unique([subscriptionId, category, city])
  @@index([category])
}
